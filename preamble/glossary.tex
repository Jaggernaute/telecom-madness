% FPGA
\newglossaryentry{fpga}{
    name={FPGA},
    description={
        A \gls{fpga_acro} is a type of configurable integrated circuit that can be repeatedly programmed after manufacturing.
    }
}

% VHDL
\newglossaryentry{vhdl}{
    name={VHDL},
    description={
        The \gls{vhdl_acro} is a hardware description language that can model the behavior and structure of digital systems at multiple levels of abstraction, ranging from the system level down to that of logic gates, for design entry, documentation, and verification purposes.
    }
}

\newacronym{fpga_acro}{FPGA}{Field Progammable Gate Array}
\newacronym{vhdl_acro}{VHDL}{\gls{vhsic_acro} Harware Description Language}
\newacronym{vhsic_acro}{VHSIC}{Very High Speed Integrated Circuit Program}
